namespace = race
scripted_trigger not_elf_race = {
	culture = culture:elvish
	NOT = { 
		OR = {
			has_trait = elf 
			has_trait = halfelf
		}
	}
}
scripted_trigger not_fertility_declined = {
	has_trait = elf
	is_female = yes
	NOT = { 
		has_trait = fertility_decline
	}
}

race.0001 = {
	hidden = yes

	trigger = {
		is_from_ruler_designer = no
		OR = {
			not_elf_race = yes
		}
	}


	immediate = {
		if = {
			limit = {
				not_elf_race = yes
			}
			add_trait = elf
		}
		if = {
			limit = {
				is_female = yes
			}
			add_trait = fertility_decline
		}
	}

}
